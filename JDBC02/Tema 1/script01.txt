/*
SCRIPT PARA CREAR VISTAS EN SQL SERVER
*/
ALTER VIEW VW_VENTAS_ANIO_ACTUAL
AS
SELECT 
	V.COD_VENTA,
	V.FEC_VENTA,
	C.COD_CLIENTE,
	C.DNI_CLIENTE,
	C.NOM_CLIENTE
FROM TA_VENTA V
INNER JOIN TA_CLIENTE C ON V.COD_CLIENTE = C.COD_CLIENTE
WHERE YEAR(V.FEC_VENTA) = YEAR(GETDATE())

SELECT * FROM VW_VENTAS_ANIO_ACTUAL


/*
LISTA DE FUNCIONES EN SQL SERVER PARA MANIPULACION DE FECHAS
*/
SELECT YEAR(GETDATE())

-- FUNCIONES PARA FECHAS
YEAR -- RETORNA EL AÑO DE UNA FECHA
MONTH -- RETORNA EL MES DE UNA FECHA 
DAY -- RETORNA EL DIA DE UNA FECHA
DATEADD -- AGREGA Y RETORNA UNA CANTIDAD DE DIAS, MESES O AÑOS A UNA FECHA
DATEDIFF -- RETORNA DIFERENCIAL ENTRE FECHAS
