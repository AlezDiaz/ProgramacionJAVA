--1. CREAR SP PARA INGRESAR AL SISTEMA VALIDANDO USUARIO Y PASSWORD

ALTER PROCEDURE USP_ACCEDER_SISTEMA
@USU_LOGUEO VARCHAR(50),
@PAS_LOGUEO VARCHAR(50)
AS
BEGIN
	SELECT COUNT(1) AS EXISTE
	FROM TA_USUARIO WHERE 
		USU_LOGUEO = @USU_LOGUEO
	AND PWDCOMPARE(@PAS_LOGUEO, PAS_LOGUEO) = 1
END

--PROBAR PROCEDIMIENTO ALMACENADO
EXEC USP_ACCEDER_SISTEMA 'LCORDOVA', '1234567'
